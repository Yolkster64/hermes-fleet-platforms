Step 1: Initialize Components
────────────────────────────
  var serviceManager = new ServerServiceManager();
  var processManager = new ProcessManager();
  var healthMonitor = new ServiceHealthMonitor(serviceManager, processManager);
  var deploymentService = new DeploymentService();

Step 2: Manage Services
────────────────────────
  var service = new ServiceInfo {
    ServiceId = "my-service",
    DisplayName = "My Service",
    AutoRestartEnabled = true,
    MaxRestartAttempts = 3
  };
  
  serviceManager.RegisterService(service);
  await serviceManager.StartServiceAsync("my-service");

Step 3: Start Health Monitoring
───────────────────────────────
  healthMonitor.Start();
  healthMonitor.HealthCheckFailed += (s, e) => 
    Console.WriteLine($"Alert: {e.ServiceId} failed");

Step 4: Deploy Applications
──────────────────────────
  var id = await deploymentService.StartDeploymentAsync(
    "MyApp", "2.0.0",
    new List<string> { "server1", "server2", "server3" },
    DeploymentType.RollingUpdate
  );
